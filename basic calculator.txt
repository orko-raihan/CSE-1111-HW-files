#include <stdio.h>

int main(){

   char operator;
   double num1;
   double num2;
   double result;

   printf("\nEnter an operator [+,-, *, / %]: ");
   scanf("%c", &operator);

 if ((operator >= 'a' && operator <= 'z'))
        printf("%c is an alphabet and not a valid operator.", operator);
    else
   printf("Enter number 1: ");
   scanf("%lf", &num1);

   printf("Enter number 2: ");
   scanf("%lf", &num2);

   switch(operator){
      case '+':
         result = num1 + num2;
         printf("\nresult: %.3lf", result);
         break;
      case '-':
         result = num1 - num2;
         printf("\nresult: %.3lf", result);
         break;
      case '*':
         result = num1 * num2;
         printf("\nresult: %.3lf", result);
         break;
      case '/':
         result = num1 / num2;
         printf("\nresult: %.3lf", result);
         break;
     default:
         printf("%c is not valid", operator);

   }

   return 0;
}
